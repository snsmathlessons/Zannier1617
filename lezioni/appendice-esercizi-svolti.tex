\chapter{Esercizi Svolti}
\justify

\newthought{Diamo nel seguito alcune tracce di soluzione degli esercizi lasciati durante il corso o referenze per vederne la risoluzione}

\vskip 2.0em
{\itshape In realtà come potete immaginare per mancanza di volontari gli esercizi non sono stati scritti}

\vskip 2.0em
\noindent Esercizi da scrivere:
\begin{itemize}
\item Ogni cubica non singolare può essere messa in forma di Weierstrass.
\item Il grado di una mappa razionale $f: \bbP^1 \rar \bbP^1$ coincide con il grado dell'estensione di campi $[\bbC(t) : \bbC(f(t))]$.
\end{itemize}
